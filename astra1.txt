<script>
    function add() {
    assign();
    document.getElementById("ans").innerHTML = (a + b);
}

function assign() {
    var n1 = document.getElementById('inp1').value;
    var n2 = document.getElementById('inp2').value;
    a = parseInt(n1);
    b = parseInt(n2);
}

function min() {
    assign();
    document.getElementById("ans").innerHTML = (a - b);
}

function mul() {
    assign();
    document.getElementById("ans").innerHTML = (a * b);
}
function div() {
    assign();
    if (b != 0)
        document.getElementById("ans").innerHTML = (a / b);
    else document.getElementById("ans").innerHTML = "Not defined";
}
